//TABLE CREATION

CREATE TABLE Parts(
Pno	VARCHAR(3),
Pname	 VARCHAR(10),
Colour	 VARCHAR(6),
PRIMARY KEY (Pno)
);

CREATE TABLE Warehouse(
Wno 	VARCHAR(3),
Wname VARCHAR(10),
Address VARCHAR(20),
PRIMARY KEY (Wno)
);

CREATE TABLE Shipments(
Pno 	VARCHAR(3),
Wno	VARCHAR(3),
Quantity NUMBER(3),
Date	DATE,
PRIMARY KEY (Pno,Wno),
FOREIGN KEY (Pno) REFERENCES Parts(Pno),
FOREIGN KEY (Wno) REFERENCES Warehouse(Wno)
);

//SQL QUERRIES

SELECT Wname
FROM Warehouse
WHERE Wno IN ( SELECT Wno
		   FROM Shipment
		   WHERE Pno = ( SELECT Pno
				  FROM Parts
				  WHERE Colour='Red'));

SELECT Pno
FROM Parts P
WHERE NOT EXISTS ( SELECT W.Wno 
		          FROM Warehouse W
		          WHERE NOT EXISTS ( SELECT  S.Wno
					      FROM Shipment S
					      WHERE S.Pno= P.Pno
						      AND S.Wno= W.Wno));

SELECT COUNT(Pno),Wno
FROM Shipment
GROUP BY (Wno);

SELECT W.Wno, W.Wname, W.Address
FROM Warehouse W JOINS Shipment S
ON W.Wno=S.Wno
WHERE Quantity IN ( SELECT MAX(Quantity)
		           FROM Shipment);
