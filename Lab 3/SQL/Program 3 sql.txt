//CREATING TABLES

CREATE TABLE Boat(
BID	VARCHAR(3),
Bname	VARCHAR(10),
Colour	VARCHAR(6),
PRIMARY KEY (BID)
);

CREATE TABLE Sailor(
SID	VARCHAR(3),
Sname	VARCHAR(10),
Age	NUMBER(2),
PRIMARY KEY (SID)
);

CREATE TABLE Reserves(
BID	VARCHAR(3),
SID	VARCHAR(3),
Day	VARCHAR(8),
PRIMARY KEY (BID,SID),
FOREIGN KEY (BID) REFERENCES Boat(BID),
FOREIGN KEY(SID) REFERNCES Sailor(SID)
);

FROM Reserves

	           FROM Sailor
	           WHERE Sname='Sami');

SELECT BID
FROM Boat B
WHERE NOT EXISTS ( SELECT S.SID 
	 	          FROM Sailor S
		           WHERE NOT EXISTS ( SELECT R.SID
					       FROM Reserves R
					       WHERE R.BID= B.BID
					       	       AND R.SID= S.SID));

SELECT COUNT(BID),SID
FROM Reserves 
GROUP BY SID;
