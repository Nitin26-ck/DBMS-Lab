//Table Creation

CREATE TABLE Supplier(
SID	VARCHAR(3),
Sname	 VARCHAR(10),
PRIMARY KEY (SID)
);

CREATE TABLE Parts(
PID	VARCHAR(3),
Pname	 VARCHAR(10),
Colour	 VARCHAR(6),
PRIMARY KEY(PID)
);

CREATE TABLE Supply(
SID	VARCHAR(3),
PID	VARCHAR(3),
Quantity  NUMBER(6),
PRIMARY KEY(SID,PID),
FOREIGN KEY (SID) REFERENCES  Supplier(SID) ON DELETE CASCADE,
FOREIGN KEY (PID) REFERENCES Parts(PID) ON DELETE CASCADE
);
